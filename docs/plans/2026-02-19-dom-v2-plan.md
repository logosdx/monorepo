# @logosdx/dom v2 Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Rewrite @logosdx/dom as an element-centric chainable DOM utility with standalone tree-shakeable functions, modern observers, AbortController lifecycle, and accessibility-first aria namespace.

**Architecture:** A `DomCollection<T>` wrapper returned by `$()` provides chainable methods. Each method delegates to standalone functions that can also be imported directly. Callable namespaces (css, attr, data, aria) act as both getter/setter functions and objects with sub-methods. All event handling integrates with `AbortController` for lifecycle management.

**Tech Stack:** TypeScript, JSDOM (tests), Vitest, @logosdx/utils

**Design doc:** `docs/plans/2026-02-19-dom-v2-design.md`

```typescript
import type { Func } from '@logosdx/utils';

/** A single element or array of elements */
export type OneOrMany<T> = T | T[];

/** CSS property names from CSSStyleDeclaration (excluding methods) */
export type CssPropNames = Extract<{
    [K in keyof CSSStyleDeclaration]: CSSStyleDeclaration[K] extends string ? K : never;
}[keyof CSSStyleDeclaration], string>;

/** CSS property values map */
export type CssProps = { [K in CssPropNames]?: CSSStyleDeclaration[K] };

/** CSS custom property name (--*) */
export type CssCustomProp = `--${string}`;

/** Any CSS property — standard or custom */
export type AnyCssProp = CssPropNames | CssCustomProp;

/** Cleanup function */
export type Cleanup = () => void;

/** Options that accept an AbortSignal */
export interface SignalOptions {
    signal?: AbortSignal;
}

/** Event types from DocumentEventMap */
export type GlobalEvents = keyof DocumentEventMap;

/** Any event name */
export type EvType = GlobalEvents | (string & {});

/** Typed event listener */
export interface EvListener<E extends EvType = EvType> {
    (ev: E extends GlobalEvents ? DocumentEventMap[E] : CustomEvent): void;
}

/** Options for $.create() */
export interface CreateOptions<E extends HTMLElement = HTMLElement> extends SignalOptions {
    text?: string;
    class?: string[];
    css?: Record<string, string>;
    attrs?: Record<string, string>;
    children?: (HTMLElement | string)[];
    on?: Record<string, EvListener>;
}

/** Options for event delegation */
export interface EventOptions extends AddEventListenerOptions, SignalOptions {
    delegate?: string;
}

/** Options for $() selection */
export interface SelectOptions extends SignalOptions {}
```

```typescript
/** Normalize to array */
export function toArray<T>(value: T | T[]): T[] {

    return Array.isArray(value) ? value : [value];
}

/** Check if a CSS property is a custom property (--*) */
export function isCustomProp(name: string): name is `--${string}` {

    return name.startsWith('--');
}

/** Convert kebab-case to camelCase */
export function kebabToCamel(str: string): string {

    return str.replace(/-([a-z])/g, (_, c) => c.toUpperCase());
}

/** Guard: only run in browser-like environment */
export function assertBrowser(name: string): void {

    if (typeof document === 'undefined') {
        throw new Error(`${name} requires a browser environment`);
    }
}
```

```typescript
import { describe, it, expect, beforeEach } from 'vitest';

const doc = window.document;

// Will import from source once created
import { css } from '../../../../packages/dom/src/css.ts';

describe('@logosdx/dom: css', () => {

    let el: HTMLDivElement;

    beforeEach(() => {
        el = doc.createElement('div');
        doc.body.appendChild(el);
    });

    describe('css() as setter', () => {

        it('should set a single CSS property', () => {

            css(el, { color: 'red' });
            expect(el.style.color).to.equal('red');
        });

        it('should set multiple CSS properties', () => {

            css(el, { color: 'red', fontSize: '16px' });
            expect(el.style.color).to.equal('red');
            expect(el.style.fontSize).to.equal('16px');
        });

        it('should set CSS custom properties', () => {

            css(el, { '--theme': 'dark' });
            expect(el.style.getPropertyValue('--theme')).to.equal('dark');
        });

        it('should apply to multiple elements', () => {

            const el2 = doc.createElement('div');
            css([el, el2], { color: 'blue' });
            expect(el.style.color).to.equal('blue');
            expect(el2.style.color).to.equal('blue');
        });
    });

    describe('css() as getter', () => {

        it('should get a single property value', () => {

            el.style.color = 'red';
            expect(css(el, 'color')).to.equal('red');
        });

        it('should get multiple property values', () => {

            el.style.color = 'red';
            el.style.fontSize = '16px';
            const result = css(el, ['color', 'fontSize']);
            expect(result).to.deep.equal({ color: 'red', fontSize: '16px' });
        });
    });

    describe('css.remove()', () => {

        it('should remove a CSS property', () => {

            el.style.color = 'red';
            css.remove(el, 'color');
            expect(el.style.color).to.equal('');
        });

        it('should remove a custom property', () => {

            el.style.setProperty('--theme', 'dark');
            css.remove(el, '--theme');
            expect(el.style.getPropertyValue('--theme')).to.equal('');
        });

        it('should remove from multiple elements', () => {

            const el2 = doc.createElement('div');
            el.style.color = 'red';
            el2.style.color = 'red';
            css.remove([el, el2], 'color');
            expect(el.style.color).to.equal('');
            expect(el2.style.color).to.equal('');
        });
    });
});
```
